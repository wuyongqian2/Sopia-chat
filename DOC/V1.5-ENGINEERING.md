from file_extractor import extract_text_only, extract_and_cache_chunks
from werkzeug.datastructures import FileStorage
from io import BytesIO


def _make_file_storage(path):
    """将磁盘文件包装为 Flask FileStorage"""
    content = path.read_bytes()
    return FileStorage(
        stream=BytesIO(content),
        filename=path.name,
        content_type="application/octet-stream"
    )


class TestExtractTextOnly:
    """聊天附件上传路径：必须返回全文，不分块"""

    def test_txt_file_returns_full_text(self, sample_txt):
        """TXT 文件应返回完整文本内容"""
        fs = _make_file_storage(sample_txt)
        result = extract_text_only(fs)
        assert result["success"] is True
        assert "测试文本" in result["text"]
        assert "第二行" in result["text"]

    def test_large_file_returns_full_text(self, large_txt):
        """大文件（>15000 字符）也必须返回全文，不能返回占位符"""
        fs = _make_file_storage(large_txt)
        result = extract_text_only(fs)
        assert result["success"] is True
        assert len(result["text"]) > 15000
        assert "文件较大" not in result["text"]  # 不应出现旧占位符

    def test_empty_file_returns_error(self, tmp_path):
        """空文件应返回错误"""
        empty = tmp_path / "empty.txt"
        empty.write_bytes(b"")
        fs = _make_file_storage(empty)
        result = extract_text_only(fs)
        assert result["success"] is False
        assert "为空" in result["error"]

    def test_unsupported_extension_returns_error(self, tmp_path):
        """不支持的文件格式应返回错误"""
        bad = tmp_path / "test.xyz"
        bad.write_bytes(b"some content")
        fs = _make_file_storage(bad)
        result = extract_text_only(fs)
        assert result["success"] is False
        assert "不支持" in result["error"]

# 三、改进项 B：模型感知上下文管理

## B-1: registry.py 添加 context_window 字段

当前 `providers/registry.py` 的 PROVIDERS 字典中，每个 provider 只有 `models`（模型名列表），没有上下文窗口信息。前端的 token 预算和 token 计数器都用了硬编码的 32000。

**改动方式**：在每个 provider 配置中新增 `context_window` 字段（单位：token 数）。这个值取该 provider 下**最小模型的上下文窗口**作为保守默认值。用户选择更大窗口的模型时，前端可以使用更宽松的预算，但保守值保证不溢出。

```python
# providers/registry.py 改动示例

PROVIDERS = {
    "deepseek": {
        "name": "DeepSeek",
        "base_url": "https://api.deepseek.com/v1",
        "models": ["deepseek-v4-pro", "deepseek-v4-flash"],
        "context_window": 65536,       # DeepSeek V4 支持 64K+
        "auth_type": "bearer",
        # ... 其余不变
    },
    "kimi": {
        "name": "Kimi (月之暗面)",
        "base_url": "https://api.moonshot.cn/v1",
        "models": ["kimi-k2.6", "kimi-k2.5", ...],
        "context_window": 8192,        # moonshot-v1-8k 是最小的
        # ...
    },
    "zhipu": {
        # ...
        "context_window": 128000,      # GLM-4 系列 128K
    },
    "qwen": {
        # ...
        "context_window": 131072,      # Qwen 系列 128K+
    },
    "ernie": {
        # ...
        "context_window": 8192,        # ernie-4.0-turbo-8k 是最小的
    },
    "doubao": {
        # ...
        "context_window": 32768,       # 豆包 Seed 2.0 32K
    },
    # ... 其余 provider 同理，按各自文档标注
}
```

**前端暴露方式**：`/api/providers` 接口（app.py 中已有）返回 PROVIDERS 字典时会自然包含 `context_window` 字段。前端 `STATE.providers` 数组中每个 provider 对象自动获得该属性，无需改后端接口。

验证：`STATE.providers.find(p => p.key === STATE.activeProvider)?.context_window` 即可获取当前 provider 的上下文窗口。

## B-2: 前端动态 token 预算计算

**当前问题**：`messageRenderer.js` 第 475 行硬编码 `MAX_FILE_TOKENS = 25000`。对所有模型一刀切——32K 模型够用但浪费，8K 模型会溢出，128K 模型严重浪费容量。

**改动文件**：`static/js/messageRenderer.js`

将 `sendMessage()` 函数中的硬编码预算替换为动态计算：

```javascript
// 替换原来的 const MAX_FILE_TOKENS = 25000;

// 动态计算文件 token 预算
const provider = STATE.providers.find(p => p.key === STATE.activeProvider);
const contextWindow = provider?.context_window || 32000;  // 回退到 32K
const systemPromptTokens = 300;   // 系统 prompt 固定开销
const outputTokens = STATE.settings.max_tokens || 4096;  // 输出预留
const historyTokens = conv.messages.length > 0
    ? conv.messages.reduce((sum, m) => sum + estimateTokens(m.content || ''), 0)
    : 0;
// 文件预算 = 上下文窗口 - 系统 prompt - 输出 - 对话历史，取 80% 作为安全余量
const MAX_FILE_TOKENS = Math.max(
    4000,  // 最低保底 4K
    Math.floor((contextWindow - systemPromptTokens - outputTokens - historyTokens) * 0.8)
);
```

**效果对比**：

| 模型上下文 | 对话历史 | 计算结果（文件预算） |
|---|---|---|
| 8K (ernie-4.0-turbo-8k) | 2K tokens | max(4000, (8192-300-4096-2000)*0.8) = 4000 |
| 32K (doubao) | 5K tokens | max(4000, (32768-300-4096-5000)*0.8) = 18697 |
| 64K (deepseek) | 10K tokens | max(4000, (65536-300-4096-10000)*0.8) = 40912 |
| 128K (kimi-k2.6) | 15K tokens | max(4000, (131072-300-4096-15000)*0.8) = 89340 |

同时修改 `conversationManager.js` 第 490 行 token 计数器的硬编码限制：

```javascript
// 替换原来的 const limit = 32000;
const provider = STATE.providers.find(p => p.key === STATE.activeProvider);
const limit = provider?.context_window || 32000;
```

## B-3: prepareMessages 改为 token-based 滑动窗口

**当前问题**：`conversationManager.js` 第 496-505 行的 `prepareMessages()` 用消息条数做滑动窗口（`maxMessages = 20`）。如果某条消息包含 5 万 token 的文件内容，20 条消息就是 100 万 token，远超任何模型的上下文窗口。

**改动方式**：在现有消息条数限制的基础上，增加 token 总量上限。两条限制取较严格的一条。

```javascript
function prepareMessages(conv) {
    const maxMessages = STATE.settings.context_messages || 20;
    const allMessages = conv.messages.map(m => ({ role: m.role, content: m.content }));

    // 消息条数限制（保留现有逻辑）
    let limited = allMessages;
    if (allMessages.length > maxMessages) {
        const firstUserIdx = allMessages.findIndex(m => m.role === 'user');
        const firstMsg = firstUserIdx >= 0 ? [allMessages[firstUserIdx]] : [];
        const recent = allMessages.slice(-maxMessages);
        limited = firstMsg.length > 0 && recent.includes(firstMsg[0]) ? recent : [...firstMsg, ...recent];
    }

    // Token 总量限制（新增）
    const provider = STATE.providers.find(p => p.key === STATE.activeProvider);
    const contextWindow = provider?.context_window || 32000;
    const outputTokens = STATE.settings.max_tokens || 4096;
    const tokenBudget = Math.floor((contextWindow - outputTokens) * 0.9);  // 留 10% 给系统 prompt

    let totalTokens = 0;
    const tokenLimited = [];
    // 从最新消息往前取，直到 token 预算用完
    for (let i = limited.length - 1; i >= 0; i--) {
        const msgTokens = estimateTokens(limited[i].content || '');
        if (totalTokens + msgTokens > tokenBudget && tokenLimited.length > 0) {
            break;  // 至少保留 1 条消息
        }
        totalTokens += msgTokens;
        tokenLimited.unshift(limited[i]);
    }

    return tokenLimited;
}
```

**设计要点**：

- 从最新消息往前取（最近对话更重要），而不是从头开始
- 至少保留 1 条消息，防止 tokenBudget 小于单条消息时返回空数组
- `* 0.9` 留 10% 给系统 prompt（约 300-500 token）
- 保留原有消息条数限制作为第一道防线，token 限制作为第二道防线
- 当 provider 无 context_window 时回退到 32000（与现有行为一致）

## 功能验证

1. 选择 DeepSeek（64K 上下文），上传一个 3 万字的文件 → 确认 toast 不弹截断提示，文件全文注入
2. 切换到文心 ernie-4.0-turbo-8k（8K 上下文），上传同样的文件 → 确认 toast 弹出截断提示，文件被截断，尾部有 `[文件内容超出上下文预算，已截断]`
3. 打开 token 计数器，确认颜色阈值随 provider 变化（不再是固定 32K）

### B 项：token-based 滑动窗口

2. 切换到 8K 模型，同样操作 → 确认只保留最近几条消息
